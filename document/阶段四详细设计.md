## 1. set_material 工具详细设计

### 1.1 功能概述

`set_material` 工具负责为场景中的对象创建和分配材质，基于 Blender 的 Principled BSDF 着色器，支持 PBR 材质属性的完整映射。

### 1.2 Principled BSDF 材质创建

```python
import bpy

def create_principled_material(
    name: str,
    base_color: tuple[float, float, float, float] = (0.8, 0.8, 0.8, 1.0),
    metallic: float = 0.0,
    roughness: float = 0.5,
    specular: float = 0.5,
    transmission: float = 0.0,
    emission_color: tuple[float, float, float, float] | None = None,
    emission_strength: float = 0.0,
    alpha: float = 1.0,
    normal_map_strength: float = 1.0,
) -> bpy.types.Material:
    """
    创建基于 Principled BSDF 的 PBR 材质。

    Args:
        name: 材质名称
        base_color: 基础颜色 (R, G, B, A)，范围 0.0-1.0
        metallic: 金属度，范围 0.0-1.0
        roughness: 粗糙度，范围 0.0-1.0
        specular: 高光强度，范围 0.0-1.0
        transmission: 透射率，范围 0.0-1.0（用于玻璃/透明材质）
        emission_color: 自发光颜色，None 表示无自发光
        emission_strength: 自发光强度
        alpha: 透明度，范围 0.0-1.0
        normal_map_strength: 法线贴图强度

    Returns:
        创建的 Blender Material 对象

    Raises:
        ValueError: 参数超出合法范围
    """
    _validate_material_params(
        base_color=base_color,
        metallic=metallic,
        roughness=roughness,
        specular=specular,
        transmission=transmission,
        alpha=alpha,
    )

    material = bpy.data.materials.new(name=name)
    material.use_nodes = True

    node_tree = material.node_tree
    principled_node = _find_principled_bsdf(node_tree)

    if principled_node is None:
        principled_node = node_tree.nodes.new('ShaderNodeBsdfPrincipled')

    principled_node.inputs['Base Color'].default_value = base_color
    principled_node.inputs['Metallic'].default_value = metallic
    principled_node.inputs['Roughness'].default_value = roughness
    principled_node.inputs['Specular'].default_value = specular
    principled_node.inputs['Transmission'].default_value = transmission
    principled_node.inputs['Alpha'].default_value = alpha

    if emission_color is not None and emission_strength > 0:
        principled_node.inputs['Emission Color'].default_value = emission_color
        principled_node.inputs['Emission Strength'].default_value = emission_strength

    if alpha < 1.0 or transmission > 0.0:
        material.blend_method = 'BLEND' if hasattr(material, 'blend_method') else None
        material.shadow_method = 'HASHED' if hasattr(material, 'shadow_method') else None

    return material

def _find_principled_bsdf(node_tree) -> bpy.types.Node | None:
    """
    在材质节点树中查找 Principled BSDF 节点。

    Args:
        node_tree: 材质节点树

    Returns:
        Principled BSDF 节点，未找到返回 None
    """
    for node in node_tree.nodes:
        if node.type == 'BSDF_PRINCIPLED':
            return node
    return None
```

### 1.4 材质分配到对象

```python
def assign_material_to_object(
    object_id: str,
    material: bpy.types.Material,
    replace: bool = False
) -> dict:
    """
    将材质分配给指定对象。

    Args:
        object_id: 目标对象 ID
        material: 材质对象
        replace: 是否替换对象上的所有现有材质（默认追加）

    Returns:
        分配结果

    Raises:
        ObjectNotFoundError: 对象不存在
        OperationNotAllowedError: 对象类型不支持材质
    """
    obj = bpy.data.objects.get(object_id)
    if obj is None:
        raise ObjectNotFoundError(f"对象 '{object_id}' 不存在")

    if obj.type not in ('MESH', 'CURVE', 'SURFACE', 'FONT', 'META'):
        raise OperationNotAllowedError(
            f"对象类型 '{obj.type}' 不支持材质分配"
        )

    if replace:
        for i in range(len(obj.material_slots)):
            obj.data.materials.pop(index=0)

    if obj.data.materials:
        obj.data.materials.append(material)
    else:
        obj.data.materials.append(material)

    slot_index = len(obj.material_slots) - 1
    obj.active_material_index = slot_index

    return {
        "success": True,
        "object_id": object_id,
        "material_name": material.name,
        "material_id": material.name,
        "slot_index": slot_index,
        "total_materials": len(obj.material_slots),
    }
```

### 1.5 材质库管理

```python
class MaterialLibrary:
    """
    材质库管理器。
    提供材质的创建、查询、复用和删除功能。
    """

    PRESETS = {
        'plastic': {
            'metallic': 0.0, 'roughness': 0.4, 'transmission': 0.0,
        },
        'glossy_plastic': {
            'metallic': 0.0, 'roughness': 0.1, 'transmission': 0.0,
        },
        'metal': {
            'metallic': 1.0, 'roughness': 0.3, 'transmission': 0.0,
        },
        'chrome': {
            'base_color': (0.8, 0.8, 0.8, 1.0),
            'metallic': 1.0, 'roughness': 0.05, 'transmission': 0.0,
        },
        'glass': {
            'metallic': 0.0, 'roughness': 0.0, 'transmission': 1.0,
        },
        'rubber': {
            'metallic': 0.0, 'roughness': 0.9, 'transmission': 0.0,
        },
        'ceramic': {
            'metallic': 0.0, 'roughness': 0.2, 'transmission': 0.0,
        },
        'wood': {
            'metallic': 0.0, 'roughness': 0.7, 'transmission': 0.0,
        },
    }

    @classmethod
    def get_or_create(cls, name: str, preset: str | None = None, **overrides) -> bpy.types.Material:
        """
        获取已有材质或创建新材质。
        如果同名材质已存在，直接返回；否则创建新材质。

        Args:
            name: 材质名称
            preset: 预设名称（可选）
            **overrides: 覆盖预设的属性值

        Returns:
            材质对象
        """
        existing = bpy.data.materials.get(name)
        if existing is not None:
            return existing

        params = {}
        if preset and preset in cls.PRESETS:
            params = dict(cls.PRESETS[preset])
        params.update(overrides)

        return create_principled_material(name=name, **params)

    @classmethod
    def list_materials(cls) -> list[dict]:
        """
        列出场景中所有材质。

        Returns:
            材质信息列表
        """
        materials = []
        for mat in bpy.data.materials:
            principled = None
            if mat.use_nodes:
                principled = _find_principled_bsdf(mat.node_tree)

            info = {
                'name': mat.name,
                'users': mat.users,
                'has_nodes': mat.use_nodes,
            }

            if principled:
                info['properties'] = {
                    'base_color': list(principled.inputs['Base Color'].default_value),
                    'metallic': principled.inputs['Metallic'].default_value,
                    'roughness': principled.inputs['Roughness'].default_value,
                    'specular': principled.inputs['Specular'].default_value,
                    'transmission': principled.inputs['Transmission'].default_value,
                }

            materials.append(info)

        return materials

    @classmethod
    def delete_material(cls, name: str) -> bool:
        """
        删除指定材质。

        Args:
            name: 材质名称

        Returns:
            是否成功删除
        """
        mat = bpy.data.materials.get(name)
        if mat is None:
            return False
        if mat.users > 0:
            return False
        bpy.data.materials.remove(mat)
        return True
```

### 1.7 参数校验

```python
def _validate_material_params(
    base_color: tuple,
    metallic: float,
    roughness: float,
    specular: float,
    transmission: float,
    alpha: float,
) -> None:
    """
    校验材质参数合法性。

    Raises:
        ValueError: 参数超出合法范围
    """
    for i, c in enumerate(base_color):
        if not (0.0 <= c <= 1.0):
            raise ValueError(
                f"base_color[{i}] = {c} 超出范围 [0.0, 1.0]"
            )

    for name, value, lo, hi in [
        ('metallic', metallic, 0.0, 1.0),
        ('roughness', roughness, 0.0, 1.0),
        ('specular', specular, 0.0, 1.0),
        ('transmission', transmission, 0.0, 1.0),
        ('alpha', alpha, 0.0, 1.0),
    ]:
        if not (lo <= value <= hi):
            raise ValueError(
                f"{name} = {value} 超出范围 [{lo}, {hi}]"
            )
```

## 2. render_scene 工具详细设计

### 2.1 功能概述

`render_scene` 工具负责配置渲染参数并触发 Blender 渲染，支持 EEVEE 和 Cycles 两种渲染引擎，支持无头模式渲染和渲染进度回调。

### 2.2 渲染引擎选择

| 引擎 | 特点 | 适用场景 | 渲染速度 | 无头模式支持 |
|------|------|---------|---------|------------|
| **EEVEE** | 实时渲染，基于光栅化 | 快速预览、风格化渲染 | 快（秒级） | ✓ |
| **Cycles** | 路径追踪，物理准确 | 照片级渲染、最终输出 | 慢（分钟级） | ✓（推荐 CPU 模式） |

```python
import bpy

RENDER_ENGINE_MAP = {
    'EEVEE': 'BLENDER_EEVEE_NEXT',
    'CYCLES': 'CYCLES',
}

def configure_render_engine(engine: str) -> None:
    """
    配置渲染引擎。

    Args:
        engine: 渲染引擎名称（'EEVEE' 或 'CYCLES'）

    Raises:
        ValueError: 不支持的渲染引擎
    """
    engine_upper = engine.upper()
    if engine_upper not in RENDER_ENGINE_MAP:
        raise ValueError(
            f"不支持的渲染引擎: '{engine}'，"
            f"可选值: {list(RENDER_ENGINE_MAP.keys())}"
        )

    bpy.context.scene.render.engine = RENDER_ENGINE_MAP[engine_upper]

    if engine_upper == 'CYCLES':
        bpy.context.scene.cycles.device = 'CPU'
        bpy.context.scene.cycles.samples = 128
        bpy.context.scene.cycles.use_denoising = True
    elif engine_upper == 'EEVEE':
        bpy.context.scene.eevee.taa_render_samples = 64
```

### 2.3 渲染参数配置

```python
def configure_render_settings(
    resolution_x: int = 1920,
    resolution_y: int = 1080,
    resolution_percentage: int = 100,
    output_path: str | None = None,
    file_format: str = 'PNG',
    color_mode: str = 'RGBA',
    color_depth: int = 8,
    compression: int = 15,
    engine: str = 'CYCLES',
    samples: int | None = None,
    use_denoising: bool = True,
) -> dict:
    """
    配置渲染参数。

    Args:
        resolution_x: 水平分辨率
        resolution_y: 垂直分辨率
        resolution_percentage: 分辨率百分比（1-100）
        output_path: 输出路径，None 则使用临时目录
        file_format: 输出格式（PNG/JPEG/TIFF/OpenEXR）
        color_mode: 颜色模式（RGB/RGBA）
        color_depth: 颜色深度（8/16/32）
        compression: 压缩级别（0-100，PNG 专用）
        engine: 渲染引擎
        samples: 采样数（None 使用引擎默认值）
        use_denoising: 是否启用降噪

    Returns:
        配置结果字典
    """
    import tempfile
    import os

    configure_render_engine(engine)

    render = bpy.context.scene.render

    render.resolution_x = resolution_x
    render.resolution_y = resolution_y
    render.resolution_percentage = resolution_percentage

    if output_path is None:
        output_dir = tempfile.mkdtemp(prefix="blender-mcp-render-")
        output_path = os.path.join(output_dir, "render.png")

    output_dir = os.path.dirname(output_path)
    if not os.path.isdir(output_dir):
        os.makedirs(output_dir, exist_ok=True)

    render.filepath = output_path

    format_map = {
        'PNG': 'PNG',
        'JPEG': 'JPEG',
        'JPG': 'JPEG',
        'TIFF': 'TIFF',
        'EXR': 'OPEN_EXR',
        'OPENEXR': 'OPEN_EXR',
    }
    render.image_settings.file_format = format_map.get(
        file_format.upper(), 'PNG'
    )
    render.image_settings.color_mode = color_mode
    render.image_settings.color_depth = str(color_depth)

    if render.image_settings.file_format == 'PNG':
        render.image_settings.compression = compression

    if samples is not None:
        if engine.upper() == 'CYCLES':
            bpy.context.scene.cycles.samples = samples
        elif engine.upper() == 'EEVEE':
            bpy.context.scene.eevee.taa_render_samples = samples

    if engine.upper() == 'CYCLES':
        bpy.context.scene.cycles.use_denoising = use_denoising

    return {
        "engine": engine,
        "resolution": [resolution_x, resolution_y],
        "resolution_percentage": resolution_percentage,
        "output_path": output_path,
        "file_format": file_format,
        "samples": samples or (
            bpy.context.scene.cycles.samples
            if engine.upper() == 'CYCLES'
            else bpy.context.scene.eevee.taa_render_samples
        ),
        "use_denoising": use_denoising,
    }
```

### 2.5 无头模式渲染

```python
def render_scene_headless(
    output_path: str | None = None,
    resolution: tuple[int, int] = (1920, 1080),
    engine: str = 'CYCLES',
    samples: int | None = None,
    use_denoising: bool = True,
) -> dict:
    """
    在无头模式下执行渲染。
    适用于 CI/CD 环境和服务器端渲染。

    Args:
        output_path: 输出路径
        resolution: 分辨率 (width, height)
        engine: 渲染引擎
        samples: 采样数
        use_denoising: 是否降噪

    Returns:
        渲染结果字典

    Raises:
        OperationNotAllowedError: 无头模式下 GPU 渲染不可用
    """
    import os
    import tempfile
    import time

    if engine.upper() == 'CYCLES':
        bpy.context.scene.cycles.device = 'CPU'

    config = configure_render_settings(
        resolution_x=resolution[0],
        resolution_y=resolution[1],
        output_path=output_path,
        engine=engine,
        samples=samples,
        use_denoising=use_denoising,
    )

    start_time = time.time()

    try:
        bpy.ops.render.render(write_still=True)
    except RuntimeError as e:
        if "GPU" in str(e) and engine.upper() == 'CYCLES':
            bpy.context.scene.cycles.device = 'CPU'
            bpy.ops.render.render(write_still=True)
        else:
            raise

    render_time = time.time() - start_time

    actual_output = bpy.context.scene.render.filepath
    if not actual_output.endswith(('.png', '.jpg', '.jpeg', '.tif', '.exr')):
        actual_output += '.png'

    file_size = 0
    if os.path.exists(actual_output):
        file_size = os.path.getsize(actual_output)

    return {
        "success": True,
        "output_path": actual_output,
        "resolution": list(resolution),
        "engine": engine,
        "render_time": round(render_time, 2),
        "file_size": file_size,
        "file_format": config["file_format"],
    }
```

### 2.6 渲染进度回调

```python
class RenderProgressCallback:
    """
    渲染进度回调处理器。
    通过 Blender 的渲染回调机制监控渲染进度，
    并通过 WebSocket 通知 MCP 客户端。
    """

    def __init__(self, notification_sender=None):
        self.notification_sender = notification_sender
        self._start_time: float = 0.0
        self._last_progress: float = 0.0
        self._progress_interval: float = 5.0

    def register(self) -> None:
        """注册渲染回调"""
        bpy.app.handlers.render_init.append(self._on_render_init)
        bpy.app.handlers.render_complete.append(self._on_render_complete)
        bpy.app.handlers.render_cancel.append(self._on_render_cancel)
        bpy.app.handlers.render_write.append(self._on_render_write)

    def unregister(self) -> None:
        """注销渲染回调"""
        for handler_list, callback in [
            (bpy.app.handlers.render_init, self._on_render_init),
            (bpy.app.handlers.render_complete, self._on_render_complete),
            (bpy.app.handlers.render_cancel, self._on_render_cancel),
            (bpy.app.handlers.render_write, self._on_render_write),
        ]:
            if callback in handler_list:
                handler_list.remove(callback)

    def _on_render_init(self, scene, depsgraph) -> None:
        """渲染开始回调"""
        import time
        self._start_time = time.time()
        self._last_progress = 0.0
        self._send_notification("render/started", {
            "engine": scene.render.engine,
            "resolution": [scene.render.resolution_x, scene.render.resolution_y],
        })

    def _on_render_write(self, scene, depsgraph) -> None:
        """渲染写入回调（可用于进度估算）"""
        import time
        now = time.time()
        if now - self._last_progress < self._progress_interval:
            return
        self._last_progress = now
        elapsed = now - self._start_time
        self._send_notification("render/progress", {
            "elapsed_seconds": round(elapsed, 1),
        })

    def _on_render_complete(self, scene, depsgraph) -> None:
        """渲染完成回调"""
        import time
        elapsed = time.time() - self._start_time
        self._send_notification("render/completed", {
            "elapsed_seconds": round(elapsed, 2),
            "output_path": scene.render.filepath,
        })

    def _on_render_cancel(self, scene, depsgraph) -> None:
        """渲染取消回调"""
        import time
        elapsed = time.time() - self._start_time
        self._send_notification("render/cancelled", {
            "elapsed_seconds": round(elapsed, 2),
        })

    def _send_notification(self, event: str, data: dict) -> None:
        """发送 MCP 通知"""
        if self.notification_sender:
            self.notification_sender.send({
                "method": "notifications/render",
                "params": {
                    "event": event,
                    "data": data,
                }
            })
```

### 2.7 参数约束

| 参数 | 类型 | 必填 | 约束 | 默认值 |
|------|------|------|------|--------|
| `output_path` | string | 否 | 必须在白名单内 | 临时目录 |
| `resolution` | int[2] | 否 | 每个维度 1-16384 | [1920, 1080] |
| `engine` | string | 否 | 枚举: `EEVEE`, `CYCLES` | `CYCLES` |
| `samples` | int | 否 | 1-4096 | 引擎默认值 |
| `use_denoising` | bool | 否 | - | true |
| `file_format` | string | 否 | 枚举: `PNG`, `JPEG`, `TIFF`, `OpenEXR` | `PNG` |

### 2.8 错误处理

| 错误场景 | 错误码 | 处理策略 |
|---------|--------|---------|
| 对象不存在 | `OBJECT_NOT_FOUND` | 返回可用对象列表 |
| 无头模式下 GPU 不可用 | `OPERATION_NOT_ALLOWED` | 自动降级为 CPU 渲染 |
| 输出目录不可写 | `PERMISSION_DENIED` | 尝试创建目录，失败则返回错误 |
| 渲染超时（> 10 分钟） | `OPERATION_TIMEOUT` | 取消渲染，返回已渲染部分 |
| 渲染过程中 Blender 崩溃 | `OPERATION_FAILED` | MCP 服务检测连接断开，标记渲染任务为 `INTERRUPTED` |
| 磁盘空间不足 | `RESOURCE_EXHAUSTED` | 渲染前检查磁盘空间 |
| 输出路径在白名单外 | `PERMISSION_DENIED` | 返回允许的路径列表 |
