### 1.3 对象命名策略

```python
class ObjectNamer:
    """对象命名管理器，处理自动命名和重名冲突。"""

    _counter: dict[str, int] = {}

    @classmethod
    def generate_name(cls, mesh_type: str, user_name: str | None = None) -> str:
        """
        生成对象名称。

        规则:
        1. 如果用户指定 name，使用用户名称；如已存在则追加 .001, .002...
        2. 如果未指定 name，使用 mesh_type 首字母大写 + 递增序号
        3. 名称仅允许 ASCII 字母数字和下划线，长度 1-64

        Args:
            mesh_type: 网格类型
            user_name: 用户指定的名称，None 则自动生成

        Returns:
            唯一的对象名称
        """
        if user_name:
            base_name = cls._sanitize_name(user_name)
        else:
            base_name = mesh_type.capitalize()

        if base_name not in bpy.data.objects:
            return base_name

        if base_name not in cls._counter:
            cls._counter[base_name] = 0

        while True:
            cls._counter[base_name] += 1
            candidate = f"{base_name}.{cls._counter[base_name]:03d}"
            if candidate not in bpy.data.objects:
                return candidate

    @staticmethod
    def _sanitize_name(name: str) -> str:
        """
        清理名称：仅保留 ASCII 字母数字和下划线，截断至 64 字符。
        """
        import re
        sanitized = re.sub(r'[^a-zA-Z0-9_]', '_', name)
        sanitized = sanitized.strip('_') or "Object"
        return sanitized[:64]
```

### 1.7 返回值结构

```python
def _build_create_result(obj: bpy.types.Object) -> dict:
    """构建 create_object 的返回值。"""
    bbox = [obj.matrix_world @ bpy.mathutils.Vector(c) for c in obj.bound_box]
    return {
        "success": True,
        "object_id": obj.name,
        "name": obj.name,
        "type": obj.type,
        "location": list(obj.location),
        "rotation": [math.degrees(a) for a in obj.rotation_euler],
        "scale": list(obj.scale),
        "vertex_count": len(obj.data.vertices) if obj.type == 'MESH' else 0,
        "face_count": len(obj.data.polygons) if obj.type == 'MESH' else 0,
        "bounds": {
            "min": [min(v[i] for v in bbox) for i in range(3)],
            "max": [max(v[i] for v in bbox) for i in range(3)],
        },
    }
```

## 2. transform_object 工具详细设计

### 2.1 absolute vs relative 模式的实现差异

| 模式 | location | rotation_euler | scale |
|------|----------|---------------|-------|
| `absolute` | `obj.location = value` | `obj.rotation_euler = [deg2rad(v) for v in value]` | `obj.scale = value` |
| `relative` | `obj.location += value` | `obj.rotation_euler[i] += deg2rad(v)` | `obj.scale[i] *= value[i]` |

### 2.2 bpy 属性映射

| MCP 参数 | bpy 属性 | 类型 | 单位转换 |
|----------|---------|------|---------|
| `location` | `obj.location` | `mathutils.Vector` | 无（直接赋值） |
| `rotation` | `obj.rotation_euler` | `mathutils.Euler` | 度 → 弧度 |
| `scale` | `obj.scale` | `mathutils.Vector` | 无（直接赋值） |

### 2.3 旋转单位转换

```python
import math

def radians_to_degrees(radians: float) -> float:
```

## 3. modify_mesh 工具详细设计

### 3.1 布尔运算实现方案

#### 3.1.1 bpy 调用方式

```python
def apply_boolean(
    object_id: str,
    target_object_id: str,
    operation: str
) -> dict:
    """
    执行布尔运算。

    Args:
        object_id: 主对象 ID
        target_object_id: 目标对象 ID
        operation: "boolean_union" | "boolean_difference" | "boolean_intersect"

    Returns:
        {"success": True, "object_id": str, "vertex_count": int, "face_count": int, "non_manifold_edges": int}
    """
    obj = bpy.data.objects[object_id]
    target = bpy.data.objects[target_object_id]

    bool_op_map = {
        "boolean_union": "UNION",
        "boolean_difference": "DIFFERENCE",
        "boolean_intersect": "INTERSECT",
    }

    modifier = obj.modifiers.new(name="Boolean", type='BOOLEAN')
    modifier.operation = bool_op_map[operation]
    modifier.object = target

    with BlenderContextAdapter.override_context():
        bpy.context.view_layer.objects.active = obj
        bpy.ops.object.modifier_apply(modifier=modifier.name)

    bpy.data.objects.remove(target, do_unlink=True)

    return _build_modify_result(obj, operation)
```

#### 3.1.2 布尔运算前置检查

```python
def _validate_boolean_operation(obj: bpy.types.Object, target: bpy.types.Object) -> list[str]:
    """
    布尔运算前的校验。

    Returns:
        警告消息列表（空列表表示无警告）
    """
    warnings = []

    if obj.type != 'MESH' or target.type != 'MESH':
        raise InvalidParameterError("布尔运算要求两个操作对象都是 MESH 类型")

    if obj.name == target.name:
        raise InvalidParameterError("布尔运算的两个对象不能是同一个")

    bbox_a = _get_world_bbox(obj)
    bbox_b = _get_world_bbox(target)

    if not _bbox_overlap(bbox_a, bbox_b):
        warnings.append(
            f"两个对象的包围盒无重叠区域。{obj.name} 包围盒: {bbox_a}, "
            f"{target.name} 包围盒: {bbox_b}。布尔运算可能无效果。"
        )

    return warnings
```

### 3.3 修改器应用 vs 保留策略

| 操作类型 | 默认策略 | 原因 |
|----------|---------|------|
| 布尔运算 | 应用修改器 | 布尔结果需要成为实际网格以供后续操作 |
| 倒角 | 应用修改器 | 倒角是永久性形状变更 |
| 挤出 | 应用修改器 | 挤出结果需要实际几何体 |
| 实体化 | 应用修改器 | 壁厚是 3D 打印的硬性要求 |

> 所有修改器默认应用。未来可通过 `apply_modifier: false` 参数保留修改器。

#### 4.1.2 bpy 调用方式

```python
def apply_simple_deform(
    object_id: str,
    operation: str,
    angle: float = 0.0,
    factor: float = 0.0,
    axis: str = "Z",
    origin: list | None = None,
    limits: list | None = None,
) -> dict:
    """
    基于 bpy.ops.object.simple_deform_modifier 的实现。

    bpy 调用:
        modifier = obj.modifiers.new(name="SimpleDeform", type='SIMPLE_DEFORM')
        modifier.deform_method = operation.upper()  # BEND/TAPER/TWIST/STRETCH
        modifier.deform_axis = axis  # 'X'/'Y'/'Z'

        if operation in ('bend', 'twist'):
            modifier.angle = math.radians(angle)
        else:
            modifier.factor = factor

        if limits:
            modifier.limits[0] = limits[0]
            modifier.limits[1] = limits[1]

        if origin:
            empty = bpy.data.objects.new("DeformOrigin", None)
            bpy.context.collection.objects.link(empty)
            empty.location = origin
            modifier.origin = empty

        bpy.ops.object.modifier_apply(modifier=modifier.name)

        if origin and empty:
            bpy.data.objects.remove(empty, do_unlink=True)
    """
```

#### 4.1.3 操作类型与参数映射

#### 4.2.2 基于 bmesh 的顶点级操作实现方案

所有雕刻操作基于 `bmesh` 模块实现，直接操作顶点数据，无需进入 Blender 的 SCULPT 模式。

```python
import bmesh
import math
from mathutils import Vector, kdtree

def _sculpt_operation(
    obj: bpy.types.Object,
    operation: str,
    center: Vector,
    radius: float,
    strength: float,
    direction: Vector | None = None,
    symmetry: str = "none",
    iterations: int = 1,
) -> dict:
    """
    执行雕刻操作的通用框架。

    流程:
    1. 获取对象的 bmesh 表示
    2. 构建 KD-Tree 加速顶点查找
    3. 查找影响半径内的顶点
    4. 对每个受影响顶点计算权重和位移
    5. 应用对称雕刻（如启用）
    6. 更新 bmesh 到对象
    """
    bm = bmesh.new()
    bm.from_mesh(obj.data)

    # 构建 KD-Tree
    kd = kdtree.KDTree(len(bm.verts))
    for i, vert in enumerate(bm.verts):
        world_pos = obj.matrix_world @ vert.co
        kd.insert(world_pos, i)
    kd.balance()

    # 查找影响半径内的顶点
    affected = []
    for (world_pos, index, dist) in kd.find_range(center, radius):
        vert = bm.verts[index]
        weight = _compute_weight(dist, radius)
        affected.append((vert, weight, world_pos, dist))

    # 执行雕刻操作
    for _ in range(iterations):
        for vert, weight, world_pos, dist in affected:
            displacement = _compute_displacement(
                operation, vert, weight, strength, direction, obj, center
            )
            vert.co += obj.matrix_world.inverted() @ displacement - obj.matrix_world.inverted() @ Vector((0,0,0))

    # 对称雕刻
    if symmetry != "none":
        _apply_symmetry(bm, obj, operation, center, radius, strength, direction, symmetry)

    bm.to_mesh(obj.data)
    bm.free()
    obj.data.update()

    return {"affected_vertices": len(affected)}
```

#### 4.2.3 Push/Pull：沿法线方向移动顶点

```python
def _compute_push_pull_displacement(
    vert, weight, strength, direction, obj, is_push: bool
) -> Vector:
    """
    Push: 沿方向/法线向外推顶点
    Pull: 沿方向/法线向内拉顶点

    位移 = weight * strength * normal_direction * sign
    - Push: sign = +1
    - Pull: sign = -1
    """
    if direction is not None:
        move_dir = direction.normalized()
    else:
        move_dir = vert.normal.normalized()

    sign = 1.0 if is_push else -1.0
    displacement = move_dir * weight * strength * sign
    return displacement
```

#### 4.2.4 Smooth：拉普拉斯平滑算法

```python
def _apply_smooth(bm, affected_verts, weight, strength, iterations=1):
    """
    拉普拉斯平滑：将每个顶点移动到其相邻顶点的加权平均位置。

    算法:
    对于每个受影响的顶点 v:
        new_pos = v.co + strength * weight * (avg_neighbor_pos - v.co)

    其中 avg_neighbor_pos 是 v 的所有相邻顶点位置的平均值。

    迭代多次可增加平滑效果，但需注意过度平滑会导致形状丢失。
    """
    for _ in range(iterations):
        new_positions = {}
        for vert, w, _, _ in affected_verts:
            if not vert.link_edges:
                continue
            neighbor_sum = Vector((0, 0, 0))
            neighbor_count = 0
            for edge in vert.link_edges:
                other = edge.other_vert(vert)
                neighbor_sum += other.co
                neighbor_count += 1
            if neighbor_count > 0:
                avg = neighbor_sum / neighbor_count
                delta = (avg - vert.co) * strength * w
                new_positions[vert] = vert.co + delta

        for vert, new_pos in new_positions.items():
            vert.co = new_pos
```

#### 4.2.5 Inflate：沿法线方向均匀外扩

```python
def _compute_inflate_displacement(vert, weight, strength) -> Vector:
    """
    Inflate: 沿顶点法线方向均匀外扩。

    与 Push 的区别:
    - Push 沿指定方向移动，方向固定
    - Inflate 沿每个顶点自身的法线方向移动，产生"膨胀"效果

    位移 = vert.normal * weight * strength
    """
    return vert.normal.normalized() * weight * strength
```

#### 4.2.6 对称雕刻：镜像轴检测和同步操作

```python
def _apply_symmetry(bm, obj, operation, center, radius, strength, direction, symmetry):
    """
    对称雕刻实现。

    流程:
    1. 根据对称轴 (x/y/z) 确定镜像变换
    2. 计算镜像中心点
    3. 在镜像区域查找受影响顶点
    4. 对镜像顶点应用相同的操作（方向取反）

    Args:
        symmetry: "x" | "y" | "z"
    """
    axis_map = {"x": 0, "y": 1, "z": 2}
    axis_idx = axis_map[symmetry]

    mirror_center = center.copy()
    mirror_center[axis_idx] = -mirror_center[axis_idx]

    mirror_direction = None
    if direction is not None:
        mirror_direction = direction.copy()
        mirror_direction[axis_idx] = -mirror_direction[axis_idx]

    kd = kdtree.KDTree(len(bm.verts))
    for i, vert in enumerate(bm.verts):
        world_pos = obj.matrix_world @ vert.co
        kd.insert(world_pos, i)
    kd.balance()

    mirror_affected = []
    for (world_pos, index, dist) in kd.find_range(mirror_center, radius):
        vert = bm.verts[index]
        weight = _compute_weight(dist, radius)
        mirror_affected.append((vert, weight, world_pos, dist))

    for vert, weight, world_pos, dist in mirror_affected:
        displacement = _compute_displacement(
            operation, vert, weight, strength, mirror_direction, obj, mirror_center
        )
        vert.co += obj.matrix_world.inverted() @ displacement
```

#### 4.3.2 衰减函数实现

```python
def _falloff(distance: float, radius: float, falloff_type: str) -> float:
    """
    计算衰减权重。

    Args:
        distance: 顶点到选择中心的距离
        radius: 选择半径
        falloff_type: 衰减类型

    Returns:
        权重值 [0.0, 1.0]，1.0 表示完全影响，0.0 表示无影响
    """
    if distance >= radius:
        return 0.0
    if distance <= 0.0:
        return 1.0

    t = distance / radius  # 归一化距离 [0, 1]

    if falloff_type == "smooth":
        return (1.0 - t) ** 2 * (1.0 + 2.0 * t)  # Hermite 插值
    elif falloff_type == "sphere":
        return math.sqrt(max(0.0, 1.0 - t * t))
    elif falloff_type == "root":
        return math.sqrt(1.0 - t)
    elif falloff_type == "sharp":
        return (1.0 - t) ** 3
    elif falloff_type == "linear":
        return 1.0 - t
    elif falloff_type == "constant":
        return 1.0
    else:
        return (1.0 - t) ** 2 * (1.0 + 2.0 * t)
```

#### 4.3.3 影响半径内的顶点选择和权重计算

```python
def _soft_transform_execute(
    obj: bpy.types.Object,
    selection_center: Vector,
    selection_radius: float,
    falloff_type: str,
    transform_type: str,
    translation: Vector | None = None,
    rotation_euler: Euler | None = None,
    scale_vec: Vector | None = None,
) -> dict:
    """
    软选择变形核心实现。

    流程:
    1. 构建 KD-Tree
    2. 查找影响半径内所有顶点
    3. 对每个顶点计算衰减权重
    4. 根据 transform_type 计算加权位移
    5. 应用位移到顶点
    """
    bm = bmesh.new()
    bm.from_mesh(obj.data)

    kd = kdtree.KDTree(len(bm.verts))
    for i, vert in enumerate(bm.verts):
        world_pos = obj.matrix_world @ vert.co
        kd.insert(world_pos, i)
    kd.balance()

    affected_count = 0
    max_displacement = 0.0

    for (world_pos, index, dist) in kd.find_range(selection_center, selection_radius):
        vert = bm.verts[index]
        weight = _falloff(dist, selection_radius, falloff_type)
        if weight < 0.001:
            continue

        affected_count += 1

        if transform_type == "translate":
            displacement_local = obj.matrix_world.inverted().to_3x3() @ translation
            vert.co += displacement_local * weight
            max_displacement = max(max_displacement, translation.length * weight)

        elif transform_type == "rotate":
            rot_matrix = rotation_euler.to_matrix()
            relative_pos = world_pos - selection_center
            rotated_pos = rot_matrix @ relative_pos
            delta_world = (rotated_pos - relative_pos) * weight
            delta_local = obj.matrix_world.inverted().to_3x3() @ delta_world
            vert.co += delta_local
            max_displacement = max(max_displacement, delta_world.length)

        elif transform_type == "scale":
            relative_pos = world_pos - selection_center
            scaled_pos = relative_pos * scale_vec
            delta_world = (scaled_pos - relative_pos) * weight
            delta_local = obj.matrix_world.inverted().to_3x3() @ delta_world
            vert.co += delta_local
            max_displacement = max(max_displacement, delta_world.length)

    bm.to_mesh(obj.data)
    bm.free()
    obj.data.update()

    return {
        "affected_vertices": affected_count,
        "max_displacement": round(max_displacement, 4),
    }
```

```python
def _check_non_manifold(obj: bpy.types.Object) -> dict:
    """
    非流形检测。

    算法:
    1. 获取 bmesh
    2. 使用 bmesh.ops.clean() 不会修改网格，仅用于分析
    3. 遍历所有边：
       - 属于 1 个面的边 → 边界边（开网格）
       - 属于 3+ 个面的边 → 非流形边
       - 属于 2 个面的边 → 正常
    4. 遍历所有顶点：
       - 连接非流形边的顶点 → 非流形顶点

    Returns:
        {"passed": bool, "non_manifold_edges": int, "boundary_edges": int, "non_manifold_vertices": int}
    """
    bm = bmesh.new()
    bm.from_mesh(obj.data)

    non_manifold_edges = 0
    boundary_edges = 0
    non_manifold_verts = set()

    for edge in bm.edges:
        face_count = len(edge.link_faces)
        if face_count == 0 or face_count > 2:
            non_manifold_edges += 1
            for vert in edge.verts:
                non_manifold_verts.add(vert.index)
        elif face_count == 1:
            boundary_edges += 1

    bm.free()

    return {
        "passed": non_manifold_edges == 0 and boundary_edges == 0,
        "non_manifold_edges": non_manifold_edges,
        "boundary_edges": boundary_edges,
        "non_manifold_vertices": len(non_manifold_verts),
    }
```

#### 5.1.3 法线检测的 bmesh 实现

```python
def _check_normals(obj: bpy.types.Object) -> dict:
    """
    法线方向检测。

    算法:
    1. 确保法线已计算
    2. 计算模型的近似质心
    3. 对每个面：
       - 计算面法线与从面中心指向质心的向量的点积
       - 如果点积 > 0，法线朝内（倒置）
    4. 统计倒置法线面数

    Returns:
        {"passed": bool, "inverted_faces": int, "total_faces": int, "inverted_ratio": float}
    """
    bm = bmesh.new()
    bm.from_mesh(obj.data)
    bm.faces.ensure_lookup_table()

    bmesh.ops.recalc_face_normals(bm, faces=bm.faces)

    center = Vector((0, 0, 0))
    for vert in bm.verts:
        center += vert.co
    center /= len(bm.verts)

    inverted_count = 0
    for face in bm.faces:
        face_center = face.calc_center_median()
        outward_dir = (face_center - center).normalized()
        dot = face.normal.dot(outward_dir)
        if dot < 0:
            inverted_count += 1

    bm.free()

    total = len(obj.data.polygons)
    return {
        "passed": inverted_count == 0,
        "inverted_faces": inverted_count,
        "total_faces": total,
        "inverted_ratio": round(inverted_count / max(total, 1), 4),
    }
```

#### 5.1.4 壁厚检测的 bmesh 实现

```python
def _check_thickness(obj: bpy.types.Object, min_thickness: float) -> dict:
    """
    壁厚检测。

    算法:
    1. 对网格的每个顶点，计算该顶点的法线方向
    2. 从该顶点沿法线反方向发射射线（使用 BVHTree）
    3. 射线与网格的最近交点距离即为该点的壁厚
    4. 若壁厚 < min_thickness，标记为壁厚不足

    注意: 此算法计算量较大，面数 > 10万时使用采样策略

    Returns:
        {"passed": bool, "thin_regions": int, "min_detected_thickness": float}
    """
    import mathutils
    from mathutils.bvhtree import BVHTree

    bm = bmesh.new()
    bm.from_mesh(obj.data)

    bvh = BVHTree.FromBMesh(bm)

    thin_count = 0
    min_detected = float('inf')
    checked = 0

    sample_step = max(1, len(bm.verts) // 10000)

    for i in range(0, len(bm.verts), sample_step):
        vert = bm.verts[i]
        checked += 1

        normal = vert.normal
        if normal.length < 0.001:
            continue

        origin = obj.matrix_world @ vert.co
        direction = -(obj.matrix_world.to_3x3() @ normal).normalized()

        location, normal_hit, index, dist = bvh.ray_cast(origin, direction, min_thickness * 2)

        if location is not None and dist < min_thickness:
            thin_count += 1
            min_detected = min(min_detected, dist)

    bm.free()

    return {
        "passed": thin_count == 0,
        "thin_regions": thin_count,
        "min_detected_thickness": round(min_detected, 4) if min_detected < float('inf') else None,
        "checked_vertices": checked,
    }
```

#### 6.1.2 基于面法线与Z轴夹角的悬垂检测算法

```python
def _detect_overhangs(
    obj: bpy.types.Object,
    angle_threshold_deg: float = 45.0
) -> dict:
    """
    悬垂检测算法。

    算法:
    1. 确定打印方向（默认 Z 轴正方向）
    2. 计算临界法线角度: critical_angle = 90 - angle_threshold
    3. 遍历所有三角面:
       a. 计算面法线
       b. 计算法线与 Z 轴的夹角
       c. 如果夹角 > critical_angle（法线指向下方超过阈值），标记为悬垂面
    4. 对相邻悬垂面进行连通区域聚类
    5. 计算每个区域的总面积和最大悬垂角度

    伪代码:
        print_direction = Vector((0, 0, 1))
        critical_angle = 90.0 - angle_threshold_deg

        for face in mesh.faces:
            normal = face.normal
            angle = math.degrees(math.acos(normal.dot(print_direction)))
            if angle > 90 + angle_threshold_deg:
                # 等价: normal.z < -sin(angle_threshold)
                标记为悬垂面
    """
    import math
    from mathutils import Vector

    print_dir = Vector((0, 0, 1))
    critical_cos = math.cos(math.radians(90 + angle_threshold_deg))

    overhang_faces = 0
    overhang_area = 0.0
    total_faces = len(obj.data.polygons)
    total_area = 0.0
    regions = []

    for poly in obj.data.polygons:
        area = poly.area
        total_area += area
        normal = poly.normal
        dot = normal.dot(print_dir)
        if dot < critical_cos:
            overhang_faces += 1
            overhang_area += area
            angle = math.degrees(math.acos(max(-1.0, min(1.0, dot))))
            center = obj.matrix_world @ poly.center
            regions.append({
                "area": round(area, 4),
                "angle": round(angle, 2),
                "position": [round(c, 4) for c in center],
            })

    severity = "none"
    ratio = overhang_area / max(total_area, 1e-10)
    if ratio < 0.05:
        severity = "low"
    elif ratio < 0.15:
        severity = "medium"
    elif ratio < 0.30:
        severity = "high"
    else:
        severity = "critical"

    return {
        "total_faces": total_faces,
        "overhang_faces": overhang_faces,
        "overhang_percentage": round(overhang_faces / max(total_faces, 1) * 100, 2),
        "overhang_regions": regions[:50],
        "requires_support": overhang_faces > 0,
        "severity": severity,
    }
```

```python
    old_api: dict,      # Blender 4.0/4.1 参数

) -> dict:
```

```python
    def handle_bpy_error(operation: str, error: Exception) -> dict:
```

#### 8.4.2 回滚机制

```python
class OperationRollback:
    """
    操作回滚管理器。
    在破坏性操作前创建快照，失败时回滚。
    """

    @staticmethod
    @contextmanager
    def snapshot_context(obj: bpy.types.Object, operation_name: str):
        """
        上下文管理器：在操作前保存对象状态，失败时回滚。

        实现:
        1. 操作前: bpy.ops.ed.undo_push(message=operation_name)
        2. 操作执行
        3. 成功: 正常退出
        4. 失败: bpy.ops.ed.undo() 回滚到操作前状态
        """
        try:
            if hasattr(bpy.ops.ed, 'undo_push'):
                bpy.ops.ed.undo_push(message=f"[before] {operation_name}")
            yield
        except Exception:
            if hasattr(bpy.ops.ed, 'undo'):
                bpy.ops.ed.undo()
            raise
```
